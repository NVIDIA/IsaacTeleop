# Runtime deps for the Wuji hand retargeter (WujiHandRetargeter, which wraps
# wuji_sdk.retargeting). The wuji-sdk[retarget] extra pulls the retargeting
# engine's own deps (numpy, scipy, nlopt, pin, pyyaml).
#
# Heads-up for aarch64: wuji-sdk[retarget] drags in nlopt. PyPI ships modern nlopt
# wheels for x86_64/macOS but not aarch64 -- there it only offers the old 2.6.2
# sdist, which won't build against NumPy 2.x. So on arm, build a current nlopt from
# source into a wheel first and install this extra with --find-links:
#
#   sudo apt-get install -y build-essential cmake swig pkg-config git
#   git clone --depth 1 --branch 2.10.0 https://github.com/DanielBok/nlopt-python /tmp/nlopt-python
#   cd /tmp/nlopt-python && git submodule update --init --recursive
#   python -m pip wheel . -w ~/.cache/nlopt-wheels   # use the target interpreter (ABI-locked .so)
#   pip install 'isaacteleop[wuji]' --find-links ~/.cache/nlopt-wheels
#
# x86_64 needs none of this (pip resolves a prebuilt manylinux nlopt wheel).

wuji-sdk[retarget]==2026.7.14

# wuji-sdk[retarget] leaves pinocchio (`pin`) unconstrained. Match isaacteleop's
# retargeter/grounding extras: the lower bound avoids pin 2.6.x wheels linked
# against older Boost sonames, while the cmeel caps keep pinocchio on the 3.9.x
# line and co-installable with [retargeters]/[grounding].
pin>=2.7.0
cmeel-urdfdom<5
cmeel-tinyxml2<11
